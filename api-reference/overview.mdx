## Quick Reference

### Basic Reward Function

```python
from reward_protocol import reward_function
from reward_protocol.models import EvaluateResult, MetricResult

@reward_function
def my_reward(messages, **kwargs) -> EvaluateResult:
    # Your evaluation logic here
    return EvaluateResult(
        score=1.0,
        reason="Evaluation successful",
        metrics={}
    )
```

### Key Data Models

```python
from reward_protocol.models import (
    EvaluateResult,     # Main evaluation result
    MetricResult,       # Individual metric result
    Message,           # Chat message structure
    ToolCall,          # Function call representation
)
```

### Evaluation Creation

```python
from reward_protocol.evaluation import create_evaluation

evaluator = create_evaluation(
    evaluator_id="my-evaluator",
    metric_folders=["metrics=./path/to/metrics"],
    display_name="My Evaluator",
    description="Custom evaluation logic"
)
```

## Common Patterns

### Error Handling

```python
@reward_function
def safe_reward(messages, **kwargs) -> EvaluateResult:
    try:
        # Your evaluation logic
        result = evaluate_response(messages)
        return EvaluateResult(score=result, reason="Success")
    except Exception as e:
        return EvaluateResult(
            score=0.0,
            reason=f"Error: {str(e)}",
            metrics={}
        )
```

### Multiple Metrics

```python
@reward_function
def multi_metric_reward(messages, **kwargs) -> EvaluateResult:
    response = messages[-1].get("content", "")
    
    # Calculate multiple metrics
    word_count = len(response.split())
    sentiment_score = analyze_sentiment(response)
    
    return EvaluateResult(
        score=(word_count * 0.3 + sentiment_score * 0.7),
        reason=f"Combined score from word count and sentiment",
        metrics={
            "word_count": MetricResult(
                score=word_count / 100.0,
                success=word_count > 10,
                reason=f"Word count: {word_count}"
            ),
            "sentiment": MetricResult(
                score=sentiment_score,
                success=sentiment_score > 0.5,
                reason=f"Sentiment score: {sentiment_score}"
            )
        }
    )
```
